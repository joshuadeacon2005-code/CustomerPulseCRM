## 🧭 What’s Happening

If you can list your projects but get no to-dos, the problem is usually one of these:

1. **Wrong endpoint path** — Basecamp 4 nests to-dos under `/buckets/{project_id}/recordings.json` rather than the old `/todosets` chain if you’re using the modern Basecamp 4 structure.
2. **Missing “todolist” tool** — Some projects may not have any to-do lists; the API will just return empty arrays.
3. **Completed to-dos hidden** — Basecamp requires a different endpoint for completed items (`/todos/completed.json`).
4. **Auth scope** — The OAuth token is valid for projects but doesn’t include the “to-dos” scope (less common, but possible).

## 🧩 Fixed Script (detects both styles)

This version:

* Tests both **modern (`/recordings.json`)** and **classic (`/todosets`)** patterns.
* Fetches *active and completed* to-dos.
* Prints raw counts and the URLs it’s hitting, so you’ll actually see if Basecamp returns empty arrays.

Paste this into Replit as `basecamp_todo_debug.py`:

```python
import os
import requests
import json
import time

ACCESS_TOKEN = os.getenv("BASECAMP_ACCESS_TOKEN")
ACCOUNT_ID = os.getenv("BASECAMP_ACCOUNT_ID")
USER_AGENT = "ReplitCRMImporter (you@example.com)"
BASE_URL = f"https://3.basecampapi.com/{ACCOUNT_ID}"

headers = {
    "Authorization": f"Bearer {ACCESS_TOKEN}",
    "User-Agent": USER_AGENT,
    "Accept": "application/json"
}

def fetch_json(url):
    while True:
        r = requests.get(url, headers=headers)
        if r.status_code == 429:
            wait = int(r.headers.get("Retry-After", "3"))
            print(f"Rate limited — waiting {wait}s")
            time.sleep(wait)
            continue
        if not r.ok:
            print(f"⚠️  {url} → {r.status_code}: {r.text[:300]}")
            return []
        return r.json()

def get_projects():
    url = f"{BASE_URL}/projects.json"
    return fetch_json(url)

def get_todo_lists(project_id):
    # Try modern Basecamp style first (recordings)
    rec_url = f"{BASE_URL}/buckets/{project_id}/recordings.json?type=TodoList"
    data = fetch_json(rec_url)
    if data:
        return data
    # Fallback for older Basecamp 3/4 patterns
    sets_url = f"{BASE_URL}/buckets/{project_id}/todosets.json"
    todo_sets = fetch_json(sets_url)
    lists = []
    for ts in todo_sets:
        lists_url = f"{BASE_URL}/buckets/{project_id}/todosets/{ts.get('id')}/todolists.json"
        lists.extend(fetch_json(lists_url))
    return lists

def get_todo_items(project_id, todolist_id):
    # active
    url_active = f"{BASE_URL}/buckets/{project_id}/todolists/{todolist_id}/todos.json"
    todos = fetch_json(url_active)
    # completed
    url_done = f"{BASE_URL}/buckets/{project_id}/todolists/{todolist_id}/todos/completed.json"
    todos += fetch_json(url_done)
    return todos

def main():
    projects = get_projects()
    print(f"✅ Found {len(projects)} projects\n")

    total = 0
    for proj in projects:
        pid = proj.get("id")
        pname = proj.get("name")
        print(f"📁 Project: {pname} ({pid})")

        lists = get_todo_lists(pid)
        print(f"  → Found {len(lists)} todo lists")
        for lst in lists:
            lid = lst.get("id")
            title = lst.get("title") or lst.get("name")
            print(f"    📋 List: {title} ({lid})")
            items = get_todo_items(pid, lid)
            print(f"      → Found {len(items)} todos in {title}")
            for t in items[:3]:  # preview first few
                print("        •", t.get("content"))
            total += len(items)
    print(f"\n🏁 Total todos fetched: {total}")

if __name__ == "__main__":
    main()
```
