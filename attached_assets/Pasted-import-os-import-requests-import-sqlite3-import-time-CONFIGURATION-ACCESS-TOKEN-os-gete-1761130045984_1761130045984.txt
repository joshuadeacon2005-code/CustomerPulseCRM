import os
import requests
import sqlite3
import time

# === CONFIGURATION ===
ACCESS_TOKEN = os.getenv("BASECAMP_ACCESS_TOKEN")
ACCOUNT_ID = os.getenv("BASECAMP_ACCOUNT_ID")
USER_AGENT = "ReplitCRMImporter (you@example.com)"  # <- change your email
BASE_URL = f"https://3.basecampapi.com/{ACCOUNT_ID}"

# SQLite database (in Replit)
DB_PATH = "crm_tasks.db"

# === DATABASE SETUP ===
def init_db():
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("""
        CREATE TABLE IF NOT EXISTS tasks (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            basecamp_todo_id INTEGER,
            project_id INTEGER,
            project_name TEXT,
            list_id INTEGER,
            list_title TEXT,
            title TEXT,
            description TEXT,
            due_date TEXT,
            completed INTEGER,
            assignees TEXT,
            UNIQUE(basecamp_todo_id)
        )
    """)
    conn.commit()
    conn.close()

def insert_task(task):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    try:
        c.execute("""
            INSERT OR IGNORE INTO tasks (
                basecamp_todo_id, project_id, project_name, list_id, list_title,
                title, description, due_date, completed, assignees
            )
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            task.get("todoId"),
            task.get("projectId"),
            task.get("projectName"),
            task.get("listId"),
            task.get("listTitle"),
            task.get("title"),
            task.get("description"),
            task.get("dueDate"),
            int(task.get("completed", False)),
            str(task.get("assignees"))
        ))
        conn.commit()
    finally:
        conn.close()

# === API HELPERS ===
headers = {
    "Authorization": f"Bearer {ACCESS_TOKEN}",
    "User-Agent": USER_AGENT,
    "Accept": "application/json"
}

def fetch_json(url):
    """Fetch JSON and handle basic rate-limiting."""
    while True:
        resp = requests.get(url, headers=headers)
        if resp.status_code == 429:  # Rate limited
            wait_time = int(resp.headers.get("Retry-After", "2"))
            print(f"Rate-limited. Waiting {wait_time}s...")
            time.sleep(wait_time)
            continue
        if not resp.ok:
            raise Exception(f"Request failed: {resp.status_code} {resp.text}")
        return resp.json(), resp.headers

def get_projects():
    url = f"{BASE_URL}/projects.json"
    data, _ = fetch_json(url)
    return data

def get_todo_sets(project_id):
    url = f"{BASE_URL}/buckets/{project_id}/todosets.json"
    data, _ = fetch_json(url)
    return data

def get_todo_lists(project_id, todoset_id):
    url = f"{BASE_URL}/buckets/{project_id}/todosets/{todoset_id}/todolists.json"
    data, _ = fetch_json(url)
    return data

def get_todo_items(project_id, todolist_id):
    # Include both active and completed todos
    url = f"{BASE_URL}/buckets/{project_id}/todolists/{todolist_id}/todos.json?completed=true"
    data, _ = fetch_json(url)
    return data

# === MAIN SCRIPT ===
def main():
    print("🔗 Connecting to Basecamp...")
    init_db()
    projects = get_projects()
    print(f"✅ Found {len(projects)} projects")

    total_tasks = 0
    for proj in projects:
        pid = proj.get("id")
        pname = proj.get("name")
        print(f"\n📁 Project: {pname} ({pid})")
        todo_sets = get_todo_sets(pid)
        if not todo_sets:
            print("  ⚠️ No todo sets found.")
            continue

        for ts in todo_sets:
            ts_id = ts.get("id")
            print(f"  🗂 TodoSet {ts_id}")
            lists = get_todo_lists(pid, ts_id)
            if not lists:
                print("    ⚠️ No to-do lists in this set.")
                continue

            for lst in lists:
                lst_id = lst.get("id")
                lst_title = lst.get("title")
                print(f"    📋 List: {lst_title} ({lst_id})")

                todos = get_todo_items(pid, lst_id)
                print(f"      → Found {len(todos)} todos")
                for todo in todos:
                    task = {
                        "todoId": todo.get("id"),
                        "projectId": pid,
                        "projectName": pname,
                        "listId": lst_id,
                        "listTitle": lst_title,
                        "title": todo.get("content"),
                        "description": todo.get("description"),
                        "dueDate": todo.get("due_on"),
                        "completed": todo.get("completed"),
                        "assignees": todo.get("assigned_to_ids"),
                    }
                    insert_task(task)
                    total_tasks += 1
    print(f"\n✅ Import complete. {total_tasks} tasks saved to {DB_PATH}.")

    # Optional: quick summary
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT COUNT(*) FROM tasks")
    count = c.fetchone()[0]
    print(f"📊 Database now contains {count} unique tasks.")
    conn.close()

if __name__ == "__main__":
    main()
